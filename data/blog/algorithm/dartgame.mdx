def solution(dartResult):
    answer = 0
    point = []
    order = -1
    pivot = False

    for i in range(len(dartResult)):

        if pivot == True:
            pivot = False
            order += 1
            continue

        if dartResult[i] >= "0" and dartResult[i] <= "9":
            if dartResult[i] == "1" and dartResult[i+1] == "0":
                point.append(10)
                pivot = True
                continue
                
            else :
                point.append(int(dartResult[i]))
                order += 1
                continue
                
        if dartResult[i] == "D" :
            point[order] = point[order] ** 2
            continue
        if dartResult[i] == "T" :
            point[order] = point[order] ** 3
            continue

        if order == 0 and dartResult[i] == "*":
            point[order] *= 2
        elif dartResult[i] == "*":
            point[order-1] *= 2
            point[order] *= 2
        if dartResult[i] == "#":
            point[order] *= -1

    for i in point:
        answer += i
            
    return answer
